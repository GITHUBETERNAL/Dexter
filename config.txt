5. Custom Service

1) cd /usr/local/bin
2) touch newservice.sh
3) vi newservice.sh
4) #!/bin/bash
5) #Service Script
while true; do 
	echo "Service running" >> /var/log/newservice.log
	sleep 10
done
:wq

6) chmod +x newservice.sh
7) vi newservice.service
[Unit]
Description = new service
After = network.target

[Service]
Type = simple
ExecStart = /usr/local/bin/newservice.sh
Restart = always
RestartSec = 5
User = root

[Install]
WantedBy=multi-user.target
:wq

8) systemctl daemon-reload
9) systemctl start newservice.service
10) systemctl statys newservice.service
11) tail -n 5 /var/log/newservice.log
